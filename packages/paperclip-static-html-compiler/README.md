Config example:

```javascript
module.exports = {
  // for all pages
  cssFileName: "index.css",
  pages: [
    {
      fileName: "index.html",
      component: {
        name: "Main"
      },
      variables: {
        "/": "home"
      }
    },
    {
      fileName: "contact.html",
      component: {
        name: "Main"
      },
      variables: {
        "/": "contact"
      }
    },
    {
      fileName: "people.html",
      data: {
        people: []
      },
      variables: {
        "/": "people"
      }
    }
  ]
};
```
